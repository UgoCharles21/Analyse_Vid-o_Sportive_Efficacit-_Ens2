    <script type="module">

        import { createClient } from 'https://cdn.jsdelivr.net'

 

        // CONFIGURATION SUPABASE (À REMPLIR)

        const SUPABASE_URL = 'https://VOTRE_ID_PROJET.supabase.co'

        const SUPABASE_KEY = 'VOTRE_CLE_ANON_PUBLIQUE'

        const supabase = createClient(SUPABASE_URL, SUPABASE_KEY)

 

        let selectedAnalysisId = null;

 

        // 1. Charger la liste des étudiants

        async function fetchData() {

            const listContainer = document.getElementById('student-list');

            const { data, error } = await supabase

                .from('analyses')

                .select('*')

                .order('created_at', { ascending: false });

 

            if (error) {

                listContainer.innerText = "Erreur de chargement : " + error.message;

                return;

            }

 

            if (data.length === 0) {

                listContainer.innerText = "Aucune analyse reçue pour le moment.";

                return;

            }

 

            listContainer.innerHTML = "";

            data.forEach(item => {

                const dateObj = new Date(item.created_at).toLocaleDateString('fr-CA');

                const div = document.createElement('div');

                div.className = "student-item";

                div.innerHTML = `<strong>${item.student_name}</strong> <span class="date">${dateObj}</span>`;

                div.onclick = () => showStudentDetails(item);

                listContainer.appendChild(div);

            });

        }

 

        // 2. Afficher les détails d'un étudiant

        function showStudentDetails(item) {

            selectedAnalysisId = item.id;

           

            // Masquer message bienvenue, afficher interface

            document.getElementById('welcome-message').style.display = "none";

            document.getElementById('correction-interface').style.display = "block";

 

            // Remplir les textes

            document.getElementById('display-name').innerText = item.student_name;

            document.getElementById('display-comment').innerText = item.student_comment || "Aucun commentaire.";

            document.getElementById('display-choices').innerText = JSON.stringify(item.choices_json, null, 2);

            document.getElementById('teacher-feedback').value = item.teacher_feedback || "";

 

            // Intégrer la vidéo YouTube

            let videoId = extractYouTubeId(item.video_url);

            if (videoId) {

                document.getElementById('video-placeholder').innerHTML = `<iframe src="https://www.youtube.com{videoId}" allowfullscreen></iframe>`;

            } else {

                document.getElementById('video-placeholder').innerHTML = "<p style='color:white; padding:20px;'>Lien vidéo invalide.</p>";

            }

        }

 

        // 3. Extraire ID YouTube

        function extractYouTubeId(url) {

            const regExp = /^.*(youtu.be\/|v\/|u\/\w\/|embed\/|watch\?v=|\&v=)([^#\&\?]*).*/;

            const match = url.match(regExp);

            return (match && match[2].length == 11) ? match[2] : null;

        }

 

        // 4. Enregistrer le feedback

        document.getElementById('save-btn').onclick = async () => {

            const feedbackText = document.getElementById('teacher-feedback').value;

           

            const { error } = await supabase

                .from('analyses')

                .update({ teacher_feedback: feedbackText })

                .eq('id', selectedAnalysisId);

 

            if (error) {

                alert("Erreur lors de la sauvegarde : " + error.message);

            } else {

                alert("Correction enregistrée avec succès !");

                fetchData(); // Rafraîchir la liste

            }

        };

 

        // Lancement initial

        window.fetchData = fetchData;

        fetchData();

    </script>
